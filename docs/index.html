But what about REAL linting?

— Yes we can!

npm i -D typescript

// tsconfig.json
{
  "compilerOptions": {
    "target": "ESNext",
    "noEmit": true,
    "allowJs": true,
    "checkJs": true,
  },
  "include": [
    "src/**/*.js"
  ]
}

// package.json
"scripts": {
  "lint": "tsc"
}
/**
 * add - Add new city
 * @param {string} word city name
 */

  //…

{ // …

      --noEmit --allowJs --checkJs"
  }
}
